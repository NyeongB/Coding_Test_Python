# formatting
a = '{0} is {num}'.format('My age', num = 29)
print(a)
# My age is 29

b = '%s is %d' % ('My age', 29)
print(b)
# My age is 29

===============================================================================
# 왼쪽 정렬
print('{0:<10}'.format('hi'))
# 가운데 정렬
print('{0:^10}'.format('hi'))
# 오른쪽 정렬
print('{0:>10}'.format('hi'))


print('{0:!<10}'.format('hi'))
print('{0:_^10}'.format('hi'))
print('{0:*>10}'.format('hi'))
===============================================================================
#소수점 표현

'{0:0.4f}.format(3.141592)'
# 3.1416
